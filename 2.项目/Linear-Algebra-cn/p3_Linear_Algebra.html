""" 
    
        x:  numpy list
        p: intNone
        infty: maxboolTruemaxFalsep

    hint:
        1.inftyTrue or False, p 
        2.intfloat
    
"""
def calc_Norm(x, p = 2, infty = False):
    x = np.array(x, dtype=np.float64)
    if infty:
        return np.max(np.abs(x))
    else:
        if p == 0:
            t = x.copy()
            t[t!=0] = 1
            return t.sum()
        elif p > 0:
            return np.power(np.sum(np.abs(x)**p), 1/p)
        else:
            return None
